# compile into assembly
g++ -std=c++20 -S main.cpp -o main.s

# compile into assembly with intel syntax
g++ -std=c++20 -S -masm=intel main.cpp -o main_intel.s

# disassembled machine code
objdump -d ./cmake-build-debug/siniminicompiler

# disassembled machine code intel style
objdump -d -M intel ./cmake-build-debug/siniminicompiler

# manual cleaner assembly without clion stuff
g++ -std=c++20 -O2 -S -masm=intel main.cpp -o main_optimized.s
